class Solution(object):
    def isValid(self, s):
        """
        :type s: str
        :rtype: bool
        """
        dic = {"(" : ")" , "{" : "}" , "[" : "]"}
        
        stack=[]

        for i in s:
            if i in dic.keys():
                stack.append(i)
                continue
            #elif (len(stack)>0 and dic[stack[-1]]==i):
            elif (stack and dic[stack[-1]]==i):
                stack.pop()
            else:
                return False
        
        if len(stack)==0:
            return True
