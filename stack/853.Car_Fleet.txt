class Solution(object):
    def carFleet(self, target, position, speed):
        """
        :type target: int
        :type position: List[int]
        :type speed: List[int]
        :rtype: int
        """
        total=[]
        for i,j in zip(position,speed):
            total.append([i,j,float(target-i)/float(j)])
        
        total.sort(reverse=True)
        
        stack = []

        for car in total:
            stack.append(car)
            if len(stack) >= 2 and  stack[-1][2]<= stack[-2][2]:
                stack.pop()
            

        return len(stack)
