class Solution(object):
    def maxArea(self, height):
        """
        :type height: List[int]
        :rtype: int
        """
        l,r = 0,len(height)-1
        a=0
        while(l<r):
            
            a = max(a, (r-l)*min(height[l],height[r]))

            if (height[l]<height[r]):
                l+=1
            else:
                r-=1
        return a
