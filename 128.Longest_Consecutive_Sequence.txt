class Solution(object):
    def longestConsecutive(self, nums):
        """
        :type nums: List[int]
        :rtype: int
        """
        s= set(nums)
        longest=0

        for n in s:
            
            if (n-1) not in s:
                length=0
                while (n+length) in s:
                    length+=1
        
                longest=max(length,longest)

        
        return longest
